> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Korean, mobile-first Fixup Studio landing page that captures free-diagnosis leads, deployable on Vercel.

**Architecture:** Next 15 App Router app mirroring `apps/dashboard`. Static server components for all 9 marketing sections (copy ported verbatim from the design handoff), 3 client components (`LeadForm`, `Faq`, `Reveal`). Lead capture goes through `POST /api/leads` → zod validation → a `LeadStore` interface with a file impl (dev) and a structured-console impl (prod fallback for Vercel's read-only FS). Shared primitives (`Button/Card/Pill/Badge`) are enriched in `packages/ui`; page-specific composites stay local.

**Tech Stack:** Next 15, React 19, TypeScript strict, Tailwind 3.4 + `@fixup/ui/tailwind-preset`, `next/font/google` (Jua/Noto Sans KR/Space Mono), zod, vitest.

**Spec:** `docs/superpowers/specs/2026-06-16-landing-design.md`

```tsx
import * as React from "react";

type Variant = "primary" | "secondary" | "ghost" | "kakao";
type Size = "md" | "lg" | "block";

export function buttonClass(variant: Variant = "primary", size: Size = "md", extra = ""): string {
  const base = "btn";
  const v = `btn-${variant}`;
  const s = size === "lg" ? "btn-lg" : size === "block" ? "btn-block" : "";
  return [base, v, s, extra].filter(Boolean).join(" ");
}

type Props = {
  variant?: Variant;
  size?: Size;
  href?: string;
  className?: string;
  children: React.ReactNode;
} & React.ButtonHTMLAttributes<HTMLButtonElement> & React.AnchorHTMLAttributes<HTMLAnchorElement>;

export function Button({ variant = "primary", size = "md", href, className = "", children, ...rest }: Props) {
  const cls = buttonClass(variant, size, className);
  if (href) return <a className={cls} href={href} {...(rest as React.AnchorHTMLAttributes<HTMLAnchorElement>)}>{children}</a>;
  return <button className={cls} {...(rest as React.ButtonHTMLAttributes<HTMLButtonElement>)}>{children}</button>;
}
```

- [ ] **Step 3: Implement `leads.ts`**

```ts
import { z } from "zod";
import type { Lead, Prospecto } from "@fixup/types";

export const zLead = z.object({
  nombre: z.string().trim().min(1),
  업체명: z.string().trim().min(1),
  rubro: z.string().trim().min(1),
  telefono: z.string().trim().regex(/^[0-9\-\s]{8,}$/),
  instagram: z.string().trim().optional(),
  naver_place: z.string().trim().optional(),
  mensaje: z.string().trim().optional(),
});
export type LeadInput = z.infer<typeof zLead>;

function slug(s: string): string {
  const base = s.trim().toLowerCase().replace(/\s+/g, "-").replace(/[^a-z0-9가-힣\-]/g, "");
  // short deterministic-ish suffix from length+codepoint sum (no Date/random at module load)
  const sum = Array.from(s).reduce((a, c) => a + c.codePointAt(0)!, 0);
  return `${base || "lead"}-${(sum % 9000) + 1000}`;
}

export function leadToProspecto(lead: Lead): Prospecto {
  const fecha = lead.creado.slice(0, 10); // YYYY-MM-DD from ISO
  return {
    id: slug(lead.업체명),
    업체명: lead.업체명,
    rubro: lead.rubro,
    instagram: lead.instagram,
    naver_place: lead.naver_place,
    telefono: lead.telefono,
    estado: "nuevo",
    observacion: lead.mensaje,
    fecha_contacto: fecha,
  };
}

export interface LeadStore {
  save(lead: Lead): Promise<{ id: string }>;
}

export class ConsoleLeadStore implements LeadStore {
  async save(lead: Lead): Promise<{ id: string }> {
    const prospecto = leadToProspecto(lead);
    console.info(JSON.stringify({ kind: "lead", lead, prospecto }));
    return { id: prospecto.id };
  }
}
```

- [ ] **Step 3: Implement `leadStore.server.ts`**

```ts
import "server-only";
import { promises as fs } from "node:fs";
import path from "node:path";
import type { Lead, Prospecto } from "@fixup/types";
import { type LeadStore, ConsoleLeadStore, leadToProspecto } from "./leads.js";

const DEFAULT_PATH =
  process.env.PROSPECTOS_PATH ?? path.join(process.cwd(), "..", "..", "data", "prospectos.json");

export class FileLeadStore implements LeadStore {
  constructor(private readonly file: string = DEFAULT_PATH) {}
  async save(lead: Lead): Promise<{ id: string }> {
    const prospecto = leadToProspecto(lead);
    const raw = await fs.readFile(this.file, "utf8");
    const parsed = JSON.parse(raw) as { prospectos?: Prospecto[] };
    const all = Array.isArray(parsed.prospectos) ? parsed.prospectos : [];
    all.push(prospecto);
    await fs.writeFile(this.file, JSON.stringify({ prospectos: all }, null, 2) + "\n", "utf8");
    return { id: prospecto.id };
  }
}

export function getLeadStore(): LeadStore {
  return process.env.LEAD_STORE === "file" ? new FileLeadStore() : new ConsoleLeadStore();
}
```

```tsx
"use client";
import * as React from "react";

export function Reveal({ children, className = "", delay = 0 }: { children: React.ReactNode; className?: string; delay?: number }) {
  const ref = React.useRef<HTMLDivElement>(null);
  React.useEffect(() => {
    document.documentElement.classList.add("js");
    const el = ref.current;
    if (!el) return;
    if (window.matchMedia("(prefers-reduced-motion: reduce)").matches) { el.classList.add("in"); return; }
    const io = new IntersectionObserver((entries) => {
      entries.forEach((e) => { if (e.isIntersecting) { el.classList.add("in"); io.unobserve(el); } });
    }, { rootMargin: "0px 0px -10% 0px" });
    io.observe(el);
    return () => io.disconnect();
  }, []);
  return <div ref={ref} className={`rv ${className}`} style={delay ? { transitionDelay: `${delay}ms` } : undefined}>{children}</div>;
}
```
